# 国公費マスタ JSONC 仕様（設計案）

> **CSV版との主な変更点**
> - 関連する複数フラグを1つのオブジェクトに統合（例：年齢上下限・端数処理・上限額）
> - 排他的な選択肢を enum 値として明示（数字コードから意味のある文字列キーへ）
> - 「パターン + 金額/割合/分数」の条件付き項目をオブジェクトで表現
> - 計算に使われない純粋なメタ情報を先頭に分離

```jsonc
{
  // ================================================================
  // ① メタ情報（計算・判定に非使用）
  // ================================================================

  "name": {
    "formal": "○○医療費助成事業",   // 必須 / 最大100文字
    "short":  "○○医療費"            // 任意 / 最大12文字
  },

  // 登録主体の識別。未入力の municipality_code が
  // 「都道府県単独事業」判定に副次的に使われるため注意。
  "jurisdiction": {
    "prefecture_code":    "13",      // 必須 / "01"〜"47"
    "municipality_code":  "131016"   // 任意 / 未指定 = 都道府県単独事業
  },

  // 政策分類番号（最大3つ、優先度の高い順に指定）
  // 分類番号は全国共通の公費適用優先順位の自動判定にも使われる。
  // 001:こども  002:ひとり親  003:感染症   004:がん
  // 005:難病・特定疾患  006:高齢者  007:妊産婦  008:寡婦
  // 009:障害者  010:不妊治療  011:被爆  012:水俣  013:その他
  "classifications": ["001"],

  "payer_number": "12345678",        // 公費負担者番号（8桁）/ 任意
  "section_code":  1,                // 同一公費負担者番号内の連番 / 必須

  "note": "..."                      // 参考情報（計算に利用されない備考）/ 任意


  // ================================================================
  // ② 適用可否の判定入力（Eligibility）
  // ================================================================

  // 制度が有効な期間。この期間外のレセプトには適用しない。
  "validity": {
    "start": "20260401",  // 必須 / YYYYMMDD / 不明の場合 "19000101"
    "end":   "99991231"   // 必須 / YYYYMMDD / 未定の場合 "99991231"
  },

  "eligibility": {

    // -- 年齢要件 --------------------------------------------------
    // basis は「何歳になったとみなす基準日」を表す。
    // 年齢要件がない場合は age キー自体を省略する。
    "age": {
      "min": {
        // "birthday"              誕生日当日
        // "prev_day"              誕生日の前日
        // "prev_day_month_start"  誕生日の前日が属する月の月初日
        // "birthday_month_start"  誕生日の属する月の月初日
        // "prev_day_fiscal_start" 誕生日の前日が属する年度の年度初日
        // "next_day"              誕生日の翌日
        // "next_month_start"      誕生日が属する月の翌月初日
        // "prev_day_next_month_start" 誕生日の前日が属する月の翌月初日
        "basis": "birthday",
        "years": 0               // この基準日で○歳以上なら対象
      },
      "max": {
        // "birthday"              誕生日当日
        // "prev_day"              誕生日の前日
        // "two_days_before"       誕生日の前々日
        // "prev_month_end"        誕生日の前日が属する月末
        // "birthday_month_end"    誕生日の属する月の月末
        // "fiscal_year_end"       誕生日の属する年度の年度末
        // "prev_day_fiscal_end"   誕生日の前日が属する年度の年度末
        "basis": "fiscal_year_end",
        "years": 18              // この基準日で○歳未満なら対象
      }
    },

    // -- 診療区分 --------------------------------------------------
    // 省略不可。どちらか一方だけでも可。
    "care_setting": ["inpatient", "outpatient"],
    // "inpatient"（入院）| "outpatient"（外来）

    // -- 保険種別 --------------------------------------------------
    "insurance_type": ["shakai", "kokuho", "kouki"],
    // "shakai"（社保）| "kokuho"（国保）| "kouki"（後期高齢）

    // -- ひとり親等の対象限定 --------------------------------------
    // ひとり親等医療費助成のみ使用。限定しない場合は省略。
    "single_parent_target": "child",
    // "child"（児童）| "guardian"（監護者）

    // -- 所得区分 --------------------------------------------------
    // 省略 = 所得要件なし
    "income": {
      "category": "low_income",
      // 標準値:
      //   "low_income" "low_income_1" "low_income_2"
      //   "livelihood_protection" "general"
      //   "working_age" "working_age_1" "working_age_2" "working_age_3"
      //   "muni_235k_under" "muni_235k_over" "muni_non_taxable"
      //   "muni_564k_under" "muni_564k_over"
      //   "middle_1" "middle_2" "general_1" "general_2" "upper_muni"
      // 独自定義: "custom_A0"〜"custom_ZZ"（下記 label が必須）
      "label": null   // 独自定義時のみ記入（メタ情報 / PMH連携用）
    },

    // -- 患者の状態 ------------------------------------------------
    // 各フラグが「該当すること」を要件とする場合 true。要件でなければ省略。
    "patient_status": {
      "high_cost_long_term": true,  // 高額かつ長期
      "severe_recognized":  false,  // 重症患者認定
      "severe_continuous":  false,  // 重度かつ継続
      "ventilator":         false   // 人工呼吸器等装着者
    },

    // -- 就学状況 --------------------------------------------------
    // 省略 = 就学状況を要件としない
    "school_status": "elementary_or_below",
    // "preschool"           未就学（小学校入学前）
    // "elementary_or_below" 小学校以下（または未就学）
    // "junior_high_or_below" 中学校以下
    // "high_school_or_below" 高校以下
    // "university_or_below"  大学など以下

    // -- 多子世帯 --------------------------------------------------
    // 省略 = 多子世帯を要件としない
    "multi_child": {
      "min_children":    3,   // 子ども○人以上の世帯が対象 / 任意
      "min_birth_order": 3    // 第○子以降が対象 / 任意
    },

    // -- 複数の属性条件の結合ロジック ------------------------------
    // 上記の属性条件（income / patient_status / school_status / multi_child 等）が
    // 複数ある場合の結合方法。条件が1つだけなら省略可。
    "condition_logic": {
      "type": "mixed",        // "AND" | "OR" | "mixed"
      // mixed の場合、AND / OR それぞれに属する項目番号リストを指定
      "and_items": ["income", "patient_status.high_cost_long_term"],
      "or_items":  ["school_status", "multi_child"]
    }

  }, // end eligibility


  // ================================================================
  // ③ 患者負担計算の入力（Calculation）
  // ================================================================

  "calculation": {

    // -- 助成方法 --------------------------------------------------
    "method": "genbutsu_receipt",
    // "genbutsu_receipt"         現物給付（レセプト併用）
    // "genbutsu_receipt_overage" 現物給付（レセプト併用）＋上限超過後は償還払い
    // "genbutsu_renkiki"         現物給付（連記式）
    // "genbutsu_renkiki_overage" 現物給付（連記式）＋上限超過後は償還払い

    // -- 助成条件 or 自己負担条件の方向 ----------------------------
    "condition_type": "subsidy",
    // "subsidy"（公費が○円/○割を負担する → 残りが患者負担）
    // "copay"  （患者が○円/○割を負担する → 残りを公費が負担）

    // -- 助成・自己負担が発生するタイミング条件 --------------------
    // 該当しない条件は省略可。
    "timing": {
      "first_visit_only":    false,  // 初診時の一部負担金のみ対象
      "first_admission_only": false, // 入院初回時の一部負担金のみ対象
      // 連続入院に関する助成内容の切り替え
      "continuous_admission": {
        "strict": true,              // true=途切れない連続入院 / false=単発入院も含める
        "start_month": 1,            // 助成内容が変わる開始月（例: 1月目から）
        "end_month":   3             // 助成内容が変わる終了月（例: 3月目まで）
      },
      // 同一月内の受診回数による絞り込み
      "visit_count": {
        "min": 4,   // ○回目以降が対象（例: 4回目以降）
        "max": null // ○回目までが対象 / 上限なしは省略
      }
    },

    // -- 助成・自己負担パターン（入院・外来 共通構造） -------------
    // pattern が "full" / "remainder" / "none" の場合、amount/ratio/fraction は省略。
    // pattern が "fixed" の場合、amount が必須。
    // pattern が "ratio_*" の場合、ratio（%）または fraction（分数）のいずれかが必須。

    "inpatient": {
      "pattern": "ratio_pct_copay",
      // "full"                全額（公費が全額負担 or 患者が全額負担）
      // "fixed"               固定額
      // "remainder"           なお残る額（自己負担額を除いた残り）
      // "ratio_pct_points"    割合%・公費対象点数ベース
      // "ratio_pct_copay"     割合%・公費適用前の一部負担金ベース
      // "ratio_pct_remainder" 割合%・なお残る額ベース
      // "ratio_frac_points"   分数・公費対象点数ベース
      // "ratio_frac_copay"    分数・公費適用前の一部負担金ベース
      // "ratio_frac_remainder" 分数・なお残る額ベース
      // "none"                なし
      "amount":   null,             // fixed 時の金額（円）
      "ratio":    10,               // ratio_pct_* 時のパーセント値（例: 10 = 10%）
      "fraction": null              // ratio_frac_* 時 { "numerator": 1, "denominator": 3 }
    },

    "outpatient": {                 // 構造は inpatient と同一
      "pattern": "fixed",
      "amount":   500,
      "ratio":    null,
      "fraction": null
    },

    // -- 食事・生活療養費の助成範囲 --------------------------------
    "meal_therapy": {
      "food_therapy": "full",
      // "full"     全額助成（患者負担なし）
      // "half"     半額助成（患者負担あり）
      // "excluded" 対象外（患者が全額負担）

      // 生活療養費は「食事療養費」と「環境療養費」を一括 or 個別に設定
      "living_therapy_mode": "combined",
      // "combined"（一括設定 → living_therapy を使用）
      // "separate"（個別設定 → living_food_therapy / living_env_therapy を使用）

      "living_therapy":      "full",      // combined 時
      "living_food_therapy": null,        // separate 時 / "full"|"half"|"excluded"
      "living_env_therapy":  null         // separate 時 / "full"|"half"|"excluded"
    },

    // -- 上限額の設定 ----------------------------------------------
    "ceiling": {

      "type": "self_burden",
      // "public_burden"（公費負担上限額：公費がここまで負担する）
      // "self_burden"  （自己負担上限額：患者がここまで払えばよい）

      // 入院と外来の上限を合算して1つの上限として扱うか
      "combine_inpatient_outpatient": false,

      // 多数回該当（同月内3回以上上限到達 → 4回目から上限額が変動）
      "multi_occurrence": {
        "inpatient":  false,
        "outpatient": false
      },

      // 端数処理
      "rounding": {
        "unit":   "1000_yen", // "1000_yen" | "100_yen" | "10_yen" | "decimal"
        "method": "floor"     // "floor"（切り捨て）| "ceil"（切り上げ）| "round"（四捨五入）
      },

      // 療養の給付・入院分
      "inpatient": {
        "per_day": {
          "amount":    3000, // 1日当たり上限額（円）
          "start_day": 1,    // 適用開始日数（入院○日目から）
          "end_day":   null  // 適用終了日数（未設定の場合は省略）
        },
        "per_month":              10000, // 1月当たり上限額（円）
        "per_month_multi":         5000  // 多数回該当時の1月当たり上限額（円）
      },

      // 療養の給付・外来分
      "outpatient": {
        "per_visit": {
          "amount":             500, // 1回当たり上限額（円）
          "max_visits_per_month": 2  // 月○回まで適用（無制限なら省略）
        },
        "per_day": {
          "amount":            500,  // 1日当たり上限額（円）
          "max_days_per_month": 10   // 月○日まで適用（無制限なら省略）
        },
        "per_month":              3000,
        "per_month_multi":        null
      },

      // 食事・生活療養費の上限額（省略可）
      "food_therapy":       { "per_day": null, "per_month": null },
      "living_therapy":     { "per_day": null, "per_month": null },
      "living_food_therapy":{ "per_day": null, "per_month": null }, // separate 時のみ
      "living_env_therapy": { "per_day": null, "per_month": null }  // separate 時のみ
    },

    // -- 負担額の累計処理 ------------------------------------------
    "carry_over": {
      // 月途中で保険者（保険制度）が変わった場合
      "on_insurance_change": "exclude",
      // "exclude"（変更前の患者負担を累計しない）
      // "include"（変更前の患者負担を累計して上限判定する）

      // 月途中で公費負担者番号が変わった場合
      "on_payer_number_change": "include_same_pref_law",
      // "exclude"                  累計しない
      // "include_same_pref_law"    同一都道府県・同一法別番号で累計
      // "include_same_pref_law_muni" 同一都道府県・同一法別番号・同一市区町村コードで累計
      // "include_with_other_laws"  同一都道府県・同一法別番号 + 累計法別番号で累計
      // "include_with_other_laws_muni" 上記 + 同一市区町村コード

      // on_payer_number_change が "include_with_other_laws*" の場合に指定（最大4件）
      "cumulative_law_numbers": ["54", "38"]
    },

    // -- 他公費の自己負担額の助成 ----------------------------------
    // 他の公費（国公費・県単・地単）の自己負担額を本制度でも助成する場合に設定。
    // 省略 = 助成しない
    "other_public_subsidy": {
      "covers_copay": "all",
      // "all"                       国公費・県単・地単の自己負担額を助成
      // "national_with_payer_number" 国公費の自己負担額を助成（法別番号採番あり）

      // 特定の公費のみを対象とする場合、法別番号または制度別番号を指定（最大3件）
      "target_businesses": ["21", "54"]
    },

    // -- 自治体内での公費適用優先順位 ------------------------------
    // 分類番号が同一の制度間でのみ有効。
    // 異なる分類番号間は共通算定モジュールが自動判定するため、その場合は 1 を指定。
    "priority": 1

  } // end calculation
}
```

## 統合・再設計のポイント

| CSV の問題 | JSONC での解決 |
|-----------|--------------|
| 年齢の「基準日」と「年齢値」が別行 | `age.min` / `age.max` オブジェクトに統合 |
| 「パターン番号」と「金額/割合/分数」が別行で条件付き | `pattern` + `amount` / `ratio` / `fraction` を1オブジェクトに |
| 食事・生活療養費が一括/個別で使うフィールドが変わる | `living_therapy_mode` で明示し、対応フィールドを分岐 |
| 上限額が入院/外来/食事/生活と種類で縦に並ぶ | `ceiling` 以下にネストし、種類を横断的に比較可能に |
| 端数処理の「単位」と「方法」が別行 | `rounding` オブジェクトに統合 |
| 多数回該当フラグ（入院/外来）が上限額から離れた位置 | `ceiling.multi_occurrence` にまとめ意味的に近接 |
| 分類番号1〜3が別行 | `classifications` 配列に統合 |
| 累計法別番号（1〜4）が別行 | `cumulative_law_numbers` 配列に統合 |
| 他事業指定（1〜3）が別行 | `target_businesses` 配列に統合 |
| 数字コードの意味が分からない | 意味のある英語キー（enum）に変換しコメントで列挙 |
